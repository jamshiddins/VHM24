    if (record.time && !moment(record.time, 'HH:mm:ss', true).isValid()) {
      recordErrors.push(`Invalid time format: ${record.time}. Expected: HH:mm:ss`);
    }

    // –°–ø–µ—Ü–∏—Ñ–∏—á–Ω–∞—è –≤–∞–ª–∏–¥–∞—Ü–∏—è –ø–æ —Ç–∏–ø—É –¥–∞–Ω–Ω—ã—Ö
